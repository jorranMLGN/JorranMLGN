Java Mastermind

                    
import java.util.Arrays;
import java.util.Scanner;

public class Main {

    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);
        int min = 0;
        int max = 5;
        int random_int;
        Integer[] answer = {0,0,0,0};
        Integer[] inputs = {0,0,0,0};

        for (int j = 0; j <= 3; j++) {
            random_int = (int) Math.floor(Math.random() * (max - min + 1) + min);
            answer[j] = random_int;
        }
//        System.out.println(Arrays.toString(answer));
        for (int j = 0; j <= 9; j++) {
            System.out.println("Yellow = 0 | Green = 1 | Purple = 2 | Red = 3 Blue  = 4 | Orange = 5");

            for (int f = 0; f <= 3; f++) {
                System.out.println("keuze voor: " + (f + 1));
                String input = scanner.nextLine();
                int intInput = Integer.parseInt(input);
                inputs[f] = intInput;
            }

            String[] roodGeel = new String[4];
            if(Arrays.equals(inputs, answer))
            {
                break;
            }
            else
            {
                for(int l = 0; l < inputs.length; l++)
                {
                    if(Arrays.asList(inputs).contains(answer[l]))
                    {
                        if(inputs[l]==(answer[l]))
                        {
                            roodGeel[l] = "wit";
                        }
                        else
                        {
                            roodGeel[l] = "zwart";
                        }
                    }
                }
                for(int k = 0; k < roodGeel.length; k++)
                {
                    System.out.println(roodGeel[k]);
                }
            }
        }
        if(Arrays.equals(inputs, answer)) {
            System.out.println("gewonnen!");
        } else {
            System.out.println("");
            System.out.println("verloren!");
        }
    }
}

const { Client, GatewayIntentBits } = require("discord.js");
require("dotenv/config");
const axios = require("axios");
let L;
let price;
let react = 0;
const client = new Client({
    intents: [
        GatewayIntentBits.Guilds,
        GatewayIntentBits.GuildMessages,
        GatewayIntentBits.MessageContent,
        GatewayIntentBits.GuildMessageReactions,
    ],
});

client.on("ready", () => {
    console.log("The bot is ready");
});

client.on("messageCreate", async (message) => {
    if (message.content === "Tanki") {
        async function getData() {
            try {
                react++;

                const response = await axios.get(
                    "https://creativecommons.tankerkoenig.de/json/list.php?lat=51.790101&lng=5.999217&rad=4&sort=price&type=e10&apikey=dbd529d9-3281-0d15-ffa8-378748fa0b3d"
                );

                const reply1 = await message.reply(
                    response.data.stations[1].name +
                        response.data.stations[1].place +
                        " €" +
                        JSON.stringify(response.data.stations[1].price)
                );

                    response.data.stations[2].name +
                        response.data.stations[2].place +

                    reply3.react("✅");

                const reply66 = await message.reply("Hoeveel liter?");

                console.log(error);

    if (react == 1) {
        if (!isNaN(message.content)) {

            reply5.react("✅");

        }

});
